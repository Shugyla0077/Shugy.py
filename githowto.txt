# 4. Коммит изменений с комментарием
$ git commit -m "Fixed header and added h1 tag"
# Теперь изменения зафиксированы в истории репозитория с комментариями.

# 5. Проверка истории коммитов
$ git log
# Показывает историю всех коммитов в репозитории, включая хеши и комментарии.

#  Пример изменения и индексации нежелательных изменений
# Добавляем нежелательный комментарий и индексируем изменения:
$ echo "<!-- This is an unwanted but staged comment -->" >> hello.html
$ git add hello.html
# Файл hello.html теперь снова проиндексирован с ненужным комментарием.

#  Отмена проиндексированных изменений (перед коммитом)
$ git restore --staged hello.html
# Отменяем изменения перед коммитом, файл вернется в состояние до индексации.
